186static int ips2_sci0_set_speed(uint32_t pclka_hz, uint32_t speed_hz)
187{
188 volatile ra4m2_sci_registers_t *sci = ips2_sci0();
189 uint32_t best_error = 0xFFFFFFFFUL;
190 uint8_t best_cks = 0U;
191 uint8_t best_brr = 0U;
192 uint8_t found = 0U;
193 uint8_t cks;
194
195 for (cks = 0U; cks < 4U; cks++)
196 {
197 uint32_t divisor = 4UL << (2U * cks);
198 uint64_t denominator = (uint64_t)speed_hz * divisor;
199 uint32_t brr_plus_one;
200 uint32_t actual_hz;
201 uint32_t error;
202
203 if (denominator == 0ULL)
204 {
205 continue;
206 }
207
208 brr_plus_one = (uint32_t)(((uint64_t)pclka_hz + (denominator / 2ULL)) /
209 denominator);
210 if ((brr_plus_one == 0U) || (brr_plus_one > 256U))
211 {
212 continue;
213 }
214
215 actual_hz = pclka_hz / (divisor * brr_plus_one);
216 error = (actual_hz > speed_hz) ?
217 (actual_hz - speed_hz) : (speed_hz - actual_hz);
218
219 if (error < best_error)
220 {
221 best_error = error;
222 best_cks = cks;
223 best_brr = (uint8_t)(brr_plus_one - 1U);
224 found = 1U;
225 }
226 }
227
228 if (found == 0U)
229 {
230 return -1;
231 }
232
233 sci->smr = (uint8_t)((sci->smr & (uint8_t)~0x03U) | best_cks);
234 sci->brr = best_brr;
235 sci->semr &= (uint8_t)~SCI_SEMR_BRME;
236 return 0;
237}

246{

248

251 (void)R_MSTP->MSTPCRB;
252
253 sci->scr = 0U;
254 sci->spmr = 0U;

762static uint32_t ips2_isqrt(uint32_t value)
763{
764 uint32_t result = 0U;
765 uint32_t bit = 1UL << 30;
766
767 while (bit > value)
768 {
769 bit >>= 2;
770 }
771
772 while (bit != 0U)
773 {
774 if (value >= result + bit)
775 {
776 value -= result + bit;
777 result = (result >> 1) + bit;
778 }
779 else
780 {
781 result >>= 1;
782 }
783 bit >>= 2;
784 }
785
786 return result;
787}

835 int32_t x1, int32_t y1,

840 int32_t dy = (y1 > y0) ? (y0 - y1) : (y1 - y0);
841 int32_t sy = (y0 < y1) ? 1 : -1;
842 int32_t error = dx + dy;
843
844 while (1)
845 {
846 if ((x0 >= 0) && (y0 >= 0) &&
847 (x0 < (int32_t)IPS2_WIDTH) && (y0 < (int32_t)IPS2_HEIGHT))

850 thickness, thickness, color);

854 {
855 break;
856 }
857
858 {
859 int32_t doubled = error * 2;
860 if (doubled >= dy)
861 {
862 error += dy;
863 x0 += sx;
864 }
865 if (doubled <= dx)
866 {
867 error += dx;
868 y0 += sy;
869 }
870 }
871 }
