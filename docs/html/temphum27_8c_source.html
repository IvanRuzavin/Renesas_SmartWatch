84static uint8_t temphum27_calculate_crc(const uint8_t *data, uint8_t length)
85{
86 uint8_t crc = 0xFFU;
87 uint8_t byte_index;
88 uint8_t bit_index;
89
90 for (byte_index = 0U; byte_index < length; byte_index++)
91 {
92 crc ^= data[byte_index];
93
94 for (bit_index = 0U; bit_index < 8U; bit_index++)
95 {
96 if ((crc & 0x80U) != 0U)
97 {
98 crc = (uint8_t)((crc << 1) ^ 0x1DU);
99 }
100 else
101 {
102 crc <<= 1;
103 }
104 }
105 }
106
107 return crc;
108}

126int temphum27_read_data(temphum27_data_t *environment)
127{
128 uint8_t command = TEMPHUM27_CMD_HOLD_TEMP_HUM;
129 uint8_t raw[5];
130 uint16_t raw_humidity;
131 uint16_t raw_temperature;
132 uint32_t humidity_tenths;
133 int32_t temperature_tenths;
134 int result;
135
136 if (environment == 0)
137 {
138 return RTC14_ERROR_ARGUMENT;
139 }
140
141 environment->valid = 0U;
142 Delay_ms(10);
143
144 result = iic0_bus_write(TEMPHUM27_I2C_ADDRESS,
145 &command, 1U, 1U);
146 if (result != RTC14_OK)
147 {
148 return result;
149 }
150
151 Delay_ms(10);
152
153 result = iic0_bus_read_after_restart(TEMPHUM27_I2C_ADDRESS,
154 raw, sizeof(raw));
155 if (result != RTC14_OK)
156 {
157 return result;
158 }
159
160 Delay_ms(10);
161
162 if (temphum27_calculate_crc(raw, 4U) != raw[4])
163 {
164 return TEMPHUM27_ERROR_CRC;
165 }
166
167 /*
168 * The Click driver returns humidity first, then temperature:
169 * RH = raw / 0x3FFF * 100
170 * T = raw / 0x3FFF * 165 - 40
171 * Keep one decimal place without requiring floating-point formatting.
172 */
173 raw_humidity = (uint16_t)(((uint16_t)raw[0] << 8) | raw[1]);
174 raw_temperature = (uint16_t)(((uint16_t)raw[2] << 8) | raw[3]);
175
176 humidity_tenths =
177 (uint32_t)((((uint64_t)raw_humidity * TEMPHUM27_HUM_SPAN_TENTHS) +
178 (TEMPHUM27_DATA_RESOLUTION / 2UL)) /
179 TEMPHUM27_DATA_RESOLUTION);
180
181 temperature_tenths =
182 (int32_t)((((uint64_t)raw_temperature *
183 (uint32_t)TEMPHUM27_TEMP_SPAN_TENTHS) +
184 (TEMPHUM27_DATA_RESOLUTION / 2UL)) /
185 TEMPHUM27_DATA_RESOLUTION) -
186 TEMPHUM27_TEMP_OFFSET_TENTHS;
187
188 if (humidity_tenths > 1000UL)
189 {
190 humidity_tenths = 1000UL;
191 }
192
193 environment->temperature_tenths_c = (int16_t)temperature_tenths;
194 environment->humidity_tenths_percent = (uint16_t)humidity_tenths;
195 environment->valid = 1U;
196
197 return RTC14_OK;
198}
