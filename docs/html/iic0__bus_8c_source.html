182static int iic0_configure_bitrate(uint32_t pclkb_hz, uint32_t bitrate_hz)
183{
184 uint8_t cks;
185 const uint8_t noise_filter_stages = 4U;
186 const uint8_t minimum_br = noise_filter_stages + 1U;
187
188 R_IIC0->ICFER |= (IIC_ICFER_SCLE | IIC_ICFER_NFE);
189 R_IIC0->ICMR3 = (uint8_t)((R_IIC0->ICMR3 & (uint8_t)~IIC_ICMR3_NF_MASK) |
190 IIC_ICMR3_NF_MASK);
191
192 for (cks = 0U; cks <= 7U; cks++)
193 {
194 uint32_t iic_clock_hz = pclkb_hz >> cks;
195 uint32_t constant = (cks == 0U) ?
196 (3U + noise_filter_stages) :
197 (2U + noise_filter_stages);
198 uint32_t clocks_per_bit;
199 uint32_t br_sum;
200 uint32_t brl;
201 uint32_t brh;
202 uint32_t actual_bitrate;
203 uint32_t error;
204
205 if (iic_clock_hz < bitrate_hz)
206 {
207 continue;
208 }
209
210 clocks_per_bit = iic_clock_hz / bitrate_hz;
211 if (clocks_per_bit <= (2U * constant))
212 {
213 continue;
214 }
215
216 br_sum = clocks_per_bit - (2U * constant);
217
218 if ((br_sum < (2U * minimum_br)) || (br_sum > 62U))
219 {
220 continue;
221 }
222
223 brl = br_sum / 2U;
224 brh = br_sum - brl;
225
226 if ((brl > 31U) || (brh > 31U))
227 {
228 continue;
229 }
230
231 actual_bitrate = iic_clock_hz / (br_sum + (2U * constant));
232 error = (actual_bitrate > bitrate_hz) ?
233 (actual_bitrate - bitrate_hz) :
234 (bitrate_hz - actual_bitrate);
235
236 /* Same 10% maximum error criterion used by the supplied SDK layer. */
237 if (((uint64_t)error * 10ULL) > bitrate_hz)
238 {
239 continue;
240 }
241
242 R_IIC0->ICMR1 = (uint8_t)(cks << IIC_ICMR1_CKS_SHIFT);
243 R_IIC0->ICBRL = (uint8_t)brl;
244 R_IIC0->ICBRH = (uint8_t)brh;
245 return RTC14_OK;
246 }
247
248 return RTC14_ERROR_CLOCK;
249}

256int iic0_bus_init(void)
257{
258 int result;
259 system_clocks_t system_clocks;
260 SYSTEM_GetClocksFrequency( &system_clocks );
261
262 /* Start IIC0: MSTPCRB.MSTPB9 = 0. */
263 R_MSTP->MSTPCRB &= ~(1UL << 9);
264 (void)R_MSTP->MSTPCRB;
265
266 /* Enter internal reset before programming IIC0. */
267 R_IIC0->ICCR1 &= (uint8_t)~IIC_ICCR1_ICE;
268 R_IIC0->ICCR1 |= IIC_ICCR1_IICRST;
269 R_IIC0->ICCR1 |= IIC_ICCR1_ICE;
270 R_IIC0->ICSER = 0U;
271
272 result = iic0_configure_bitrate(system_clocks.pclkb, APP_I2C_BITRATE_HZ);
273 if (result != RTC14_OK)
274 {
275 return result;
276 }
277
278 /* Release internal reset and connect the selected peripheral pins. */
279 R_IIC0->ICCR1 &= (uint8_t)~IIC_ICCR1_IICRST;
280 iic0_configure_pins();
281 iic0_clear_receive_state();
282
283 return RTC14_OK;
284}

299int iic0_bus_write(uint8_t address, const uint8_t *data,
300 uint32_t length, uint8_t restart_after_write)
301{
302 uint32_t index;
303 int result;
304
305 if ((data == 0) || (length == 0U))
306 {
307 return RTC14_ERROR_ARGUMENT;
308 }
309
310 result = iic0_wait_clear(&R_IIC0->ICCR2, IIC_ICCR2_BBSY);
311 if (result != RTC14_OK)
312 {
313 return result;
314 }
315
316 R_IIC0->ICSR2 &= (uint8_t)~(IIC_ICSR2_NACKF | IIC_ICSR2_STOP);
317 R_IIC0->ICCR2 |= IIC_ICCR2_ST;
318
319 result = iic0_wait_set(&R_IIC0->ICSR2, IIC_ICSR2_TDRE);
320 if (result != RTC14_OK)
321 {
322 return result;
323 }
324
325 R_IIC0->ICDRT = (uint8_t)(address << 1);
326
327 result = iic0_wait_transmit_flag(IIC_ICSR2_TDRE);
328 if (result != RTC14_OK)
329 {
330 (void)iic0_request_stop();
331 return result;
332 }
333
334 for (index = 0U; index < length; index++)
335 {
336 result = iic0_wait_set(&R_IIC0->ICSR2, IIC_ICSR2_TDRE);
337 if (result != RTC14_OK)
338 {
339 return result;
340 }
341
342 R_IIC0->ICDRT = data[index];
343 }
344
345 result = iic0_wait_transmit_flag(IIC_ICSR2_TEND);
346 if (result != RTC14_OK)
347 {
348 (void)iic0_request_stop();
349 return result;
350 }
351
352 if (restart_after_write != 0U)
353 {
354 R_IIC0->ICCR2 |= IIC_ICCR2_RS;
355 return RTC14_OK;
356 }
357
358 return iic0_request_stop();
359}
