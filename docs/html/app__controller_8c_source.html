728void app_controller_run(void)
729{
730 while (1)
731 {
732 uint8_t did_work = 0U;
733 uint32_t now = app_system_ms;
734
735 did_work |= bletiny_process_rx();
736 did_work |= app_handle_ble_status();
737 did_work |= bletiny_service(now);
738 did_work |= app_handle_auto_b3(now);
739 did_work |= app_handle_physical_b3(now);
740 did_work |= app_handle_physical_b4();
741 did_work |= app_handle_ble_screen_request(now);
742 did_work |= app_handle_ble_stopwatch_reset(now);
743 did_work |= app_handle_ble_stopwatch_control(now);
744 did_work |= app_handle_periodic_update(now);
745 did_work |= app_refresh_active_screen_if_due(now);
746
747 if (did_work == 0U)
748 {
749 __WFI();
750 }
751 }
