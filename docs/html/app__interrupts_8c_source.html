32static void rtc14_install_interrupt_vectors(void)
33{
34 uint32_t index;
35 uint32_t primask = __get_PRIMASK();
36 const uintptr_t *active_vector_table =
37 (const uintptr_t *)(uintptr_t)SCB->VTOR;
38
39 __disable_irq();
40
41 for (index = 0U; index < APP_VECTOR_COUNT; index++)
42 {
43 app_ram_vector_table[index] = active_vector_table[index];
44 }
45
46 app_ram_vector_table[APP_CORE_VECTOR_COUNT + APP_AGT0_IRQ_SLOT] =
47 (uintptr_t)AGT0_IRQHandler;
48 app_ram_vector_table[APP_CORE_VECTOR_COUNT + APP_B3_IRQ_SLOT] =
49 (uintptr_t)P304_IRQ9_IRQHandler;
50 app_ram_vector_table[APP_CORE_VECTOR_COUNT + APP_B4_IRQ_SLOT] =
51 (uintptr_t)P111_IRQ4_IRQHandler;
52 app_ram_vector_table[APP_CORE_VECTOR_COUNT + BLETINY_SCI3_RX_IRQ_SLOT] =
53 (uintptr_t)BLETINY_SCI3_RXI_IRQHandler;
54 app_ram_vector_table[APP_CORE_VECTOR_COUNT + BLETINY_SCI3_ERI_IRQ_SLOT] =
55 (uintptr_t)BLETINY_SCI3_ERI_IRQHandler;
56
57 SCB->VTOR = (uint32_t)(uintptr_t)app_ram_vector_table;
58 __DSB();
59 __ISB();
60
61 if ((primask & 1U) == 0U)
62 {
63 __enable_irq();
64 }
65}

75static int rtc14_agt0_select_1ms_clock(uint32_t pclkb_hz,
76 uint8_t *agtmr1,
77 uint16_t *reload)
78{
79 static const uint8_t tck_value[3] =
80 {
81 AGT_AGTMR1_TCK_PCLKB_DIV_1,
82 AGT_AGTMR1_TCK_PCLKB_DIV_2,
83 AGT_AGTMR1_TCK_PCLKB_DIV_8
84 };
85 static const uint8_t divisor[3] = { 1U, 2U, 8U };
86 uint32_t index;
87
88 if ((agtmr1 == 0) || (reload == 0) || (pclkb_hz == 0U))
89 {
90 return RTC14_ERROR_ARGUMENT;
91 }
92
93 for (index = 0U; index < 3U; index++)
94 {
95 uint32_t source_hz = pclkb_hz / divisor[index];
96 uint32_t counts = (source_hz + (APP_AGT0_TICK_HZ / 2U)) /
97 APP_AGT0_TICK_HZ;
98
99 if ((counts != 0U) && (counts <= 65535U))
100 {
101 *agtmr1 = tck_value[index];
102 *reload = (uint16_t)(counts - 1U);
103 return RTC14_OK;
104 }
105 }
106
107 return RTC14_ERROR_CLOCK;
108}

116static int rtc14_agt0_init(uint32_t pclkb_hz)
117{
118 volatile ra4m2_agt16_registers_t *agt = rtc14_agt0_registers();
119 uint32_t timeout = APP_I2C_TIMEOUT;
120 IRQn_Type irq = (IRQn_Type)APP_AGT0_IRQ_SLOT;
121 uint8_t agtmr1;
122 uint16_t reload;
123 int result;
124
125 result = rtc14_agt0_select_1ms_clock(pclkb_hz, &agtmr1, &reload);
126 if (result != RTC14_OK)
127 {
128 return result;
129 }
130
131 NVIC_DisableIRQ(irq);
132 NVIC_ClearPendingIRQ(irq);
133 R_ICU->IELSR[APP_AGT0_IRQ_SLOT] = 0U;
134
135 /* Start AGT0: MSTPCRD.MSTPD3 = 0. */
136 R_MSTP->MSTPCRD &= ~(1UL << AGT0_MSTPCRD_BIT);
137 (void)R_MSTP->MSTPCRD;
138
139 agt->AGTCR = 0U;
140 while ((agt->AGTCR & AGT_AGTCR_TCSTF) != 0U)
141 {
142 if (timeout-- == 0U)
143 {
144 return RTC14_ERROR_TIMEOUT;
145 }
146 }
147
148 /* Periodic mode using PCLKB / 1, /2 or /8 for an accurate 1 ms tick. */
149 agt->AGTMR2 = 0U;
150 agt->AGTMR1 = agtmr1;
151 agt->AGTIOC = 0U;
152 agt->AGTISR = 0U;
153 agt->AGTCMSR = 0U;
154 agt->AGTIOSEL = 0U;
155 agt->AGTCMA = 0U;
156 agt->AGTCMB = 0U;
157 agt->AGT = reload;
158
159 R_ICU->IELSR[APP_AGT0_IRQ_SLOT] = APP_AGT0_EVENT;
160
161 NVIC_SetPriority(irq, APP_AGT0_IRQ_PRIORITY);
162 NVIC_ClearPendingIRQ(irq);
163 NVIC_EnableIRQ(irq);
164
165 agt->AGTCR = AGT_AGTCR_START_TIMER;
166
167 timeout = APP_I2C_TIMEOUT;
168 while ((agt->AGTCR & AGT_AGTCR_TCSTF) == 0U)
169 {
170 if (timeout-- == 0U)
171 {
172 NVIC_DisableIRQ(irq);
173 return RTC14_ERROR_TIMEOUT;
174 }
175 }
176
177 return RTC14_OK;
178}

184static void rtc14_p304_irq_init(void)
185{
186 IRQn_Type irq = (IRQn_Type)APP_B3_IRQ_SLOT;
187
188 NVIC_DisableIRQ(irq);
189 NVIC_ClearPendingIRQ(irq);
190 R_ICU->IELSR[APP_B3_IRQ_SLOT] = 0U;
191
192 /* P304 as IRQ9 input, active-low button with internal pull-up. */
193 R_PMISC->PWPR_b.B0WI = 0U;
194 R_PMISC->PWPR_b.PFSWE = 1U;
195 R_PFS->PORT[3].PIN[4].PmnPFS = PFS_PCR_PULLUP | PFS_ISEL_IRQ;
196 R_PMISC->PWPR_b.PFSWE = 0U;
197 R_PMISC->PWPR_b.B0WI = 1U;
198
199 /* IRQ9 falling edge. IELSR must be disconnected while IRQCR is changed. */
200 R_ICU->IRQCR[APP_B3_IRQ_CHANNEL] = ICU_IRQCR_FALLING_EDGE;
201 R_ICU->IELSR[APP_B3_IRQ_SLOT] = APP_B3_IRQ_EVENT;
202
203 NVIC_SetPriority(irq, APP_B3_IRQ_PRIORITY);
204 NVIC_ClearPendingIRQ(irq);
205 NVIC_EnableIRQ(irq);
206}
