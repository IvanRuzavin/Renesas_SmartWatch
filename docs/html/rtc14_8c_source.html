141int rtc14_set_datetime_24h(uint16_t year, uint8_t month,
142 uint8_t day, uint8_t hour,
143 uint8_t minute, uint8_t second,
144 uint8_t weekday)
145{
146 uint8_t frame[8];
147 uint8_t hour_12;
148 uint8_t pm = 0U;
149 uint8_t maximum_day;
150
151 maximum_day = rtc14_days_in_month(year, month);
152 if ((year < 2000U) || (year > 2099U) ||
153 (maximum_day == 0U) || (day == 0U) || (day > maximum_day) ||
154 (hour > 23U) || (minute > 59U) || (second > 59U) ||
155 (weekday > 6U))
156 {
157 return RTC14_ERROR_ARGUMENT;
158 }
159
160 /* Convert the 24-hour BLE command to the RTC's 12-hour AM/PM format. */
161 if (hour == 0U)
162 {
163 hour_12 = 12U;
164 }
165 else if (hour < 12U)
166 {
167 hour_12 = hour;
168 }
169 else if (hour == 12U)
170 {
171 hour_12 = 12U;
172 pm = 1U;
173 }
174 else
175 {
176 hour_12 = (uint8_t)(hour - 12U);
177 pm = 1U;
178 }
179
180 frame[0] = RTC14_REG_SECONDS;
181 frame[1] = rtc14_decimal_to_bcd(second);
182 frame[2] = rtc14_decimal_to_bcd(minute);
183 frame[3] = (uint8_t)(RTC14_HOUR_FORMAT_12 |
184 ((pm != 0U) ? RTC14_HOUR_PM : 0U) |
185 rtc14_decimal_to_bcd(hour_12));
186 frame[4] = rtc14_decimal_to_bcd(day);
187 frame[5] = rtc14_decimal_to_bcd(month);
188 frame[6] = rtc14_decimal_to_bcd((uint8_t)(year - 2000U));
189 frame[7] = rtc14_decimal_to_bcd(weekday);
190
191 return iic0_bus_write(RTC14_I2C_ADDRESS,
192 frame, sizeof(frame), 0U);
193}

217int rtc14_get_datetime(rtc14_datetime_t *datetime)
218{
219 uint8_t raw[7];
220 uint8_t raw_hour;
221 int result;
222
223 if (datetime == 0)
224 {
225 return RTC14_ERROR_ARGUMENT;
226 }
227
228 /* Read 00h..06h in one transaction so date and time form one snapshot. */
229 result = rtc14_read_registers(RTC14_REG_SECONDS, raw, 7U);
230 if (result != RTC14_OK)
231 {
232 return result;
233 }
234
235 datetime->second = rtc14_bcd_to_decimal(raw[0]);
236 datetime->minute = rtc14_bcd_to_decimal(raw[1]);
237
238 raw_hour = raw[2];
239 datetime->is_24_hour = ((raw_hour & RTC14_HOUR_FORMAT_24) != 0U) ? 1U : 0U;
240 datetime->is_pm = 0U;
241
242 if (datetime->is_24_hour != 0U)
243 {
244 raw_hour &= RTC14_HOUR_24_MASK;
245 }
246 else
247 {
248 datetime->is_pm = ((raw_hour & RTC14_HOUR_PM) != 0U) ? 1U : 0U;
249 raw_hour &= RTC14_HOUR_12_MASK;
250 }
251
252 datetime->hour = rtc14_bcd_to_decimal(raw_hour);
253 datetime->day = rtc14_bcd_to_decimal(raw[3]);
254 datetime->month = rtc14_bcd_to_decimal(raw[4]);
255 datetime->year = rtc14_bcd_to_decimal(raw[5]);
256 datetime->weekday = rtc14_bcd_to_decimal(raw[6]);
257
258 return RTC14_OK;
259}
